from Crypto.Cipher import DES from Crypto.Util.Padding import pad, unpad from Crypto.Random import get_random_bytes # Function to encrypt data using DES def des_encrypt(plaintext, key): # Create a DES cipher object cipher = DES.new(key, DES.MODE_ECB) # Pad the plaintext to be a multiple of 8 bytes (block size for DES) padded_text = pad(plaintext.encode('utf-8'), DES.block_size) # Encrypt the padded plaintext ciphertext = cipher.encrypt(padded_text) return ciphertext # Function to decrypt data using DES def des_decrypt(ciphertext, key): # Create a DES cipher object cipher = DES.new(key, DES.MODE_ECB) # Decrypt the ciphertext decrypted_padded_text = cipher.decrypt(ciphertext) # Unpad the decrypted plaintext to get the original message plaintext = unpad(decrypted_padded_text, DES.block_size).decode('utf-8') return plaintext # Example usage if _name_ == "_main_": try: # 8-byte key for DES (64 bits, but only 56 bits are used, 8 bits are for parity) key = get_random_bytes(8) # Plaintext message plaintext = "Hello, World!" # Encrypt the plaintext ciphertext = des_encrypt(plaintext, key) print(f"Ciphertext: {ciphertext.hex()}") # Decrypt the ciphertext decrypted_text = des_decrypt(ciphertext, key) print(f"Decrypted text: {decrypted_text}") except Exception as e: print(f"An error occurred: {e}")
